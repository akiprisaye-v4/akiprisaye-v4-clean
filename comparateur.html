/* ================================ COMPARATEUR – LOGIQUE PRINCIPALE ================================ */ document.addEventListener("DOMContentLoaded", () => { const searchInput = document.getElementById("searchInput"); const refreshBtn = document.getElementById("refreshBtn"); const resultsList = document.getElementById("resultsList"); // ▼ Données simulées (reliées à comparateur-data.js) let PRODUCTS = window.PRODUCTS || []; /* ------------------------------ 1. Recherche en temps réel ------------------------------ */ searchInput.addEventListener("input", (e) => { const term = e.target.value.toLowerCase().trim(); displayResults(filterProducts(term)); }); /* ------------------------------ 2. Bouton "Actualiser les prix" ------------------------------ */ refreshBtn.addEventListener("click", () => { refreshBtn.classList.add("loading"); refreshBtn.disabled = true; // Ici, plus tard → future synchro Firestore / API setTimeout(() => { refreshBtn.classList.remove("loading"); refreshBtn.disabled = false; displayResults(filterProducts(searchInput.value.toLowerCase())); }, 800); }); /* ------------------------------ 3. Filtrer les produits ------------------------------ */ function filterProducts(term) { if (!term) return PRODUCTS; return PRODUCTS.filter(p => p.name.toLowerCase().includes(term) || p.brand.toLowerCase().includes(term) ); } /* ------------------------------ 4. Afficher les résultats ------------------------------ */ function displayResults(list) { resultsList.innerHTML = ""; if (list.length === 0) { resultsList.innerHTML = `
Aucun produit trouvé…
`; return; } list.forEach(item => { const card = document.createElement("div"); card.className = "product-card"; card.innerHTML = `

${item.name}

${item.brand}

${item.price.toFixed(2)} € ${item.store} ${item.territory}
`; resultsList.appendChild(card); }); } /* ------------------------------ 5. Initialisation ------------------------------ */ displayResults(PRODUCTS); });
